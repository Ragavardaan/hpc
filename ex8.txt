#include <stdio.h>
#include <omp.h>

int main() {
    const int H = 4, W = 6;
    int img[H][W];
    printf("----------------------- To enhance the quality of image pixels using parallel for----------------------------\n");

    // fill sample values
    for (int i = 0; i < H; i++)
        for (int j = 0; j < W; j++)
            img[i][j] = i * W + j;
    
    printf("Pixel values before Operation:\n");
    for (int i = 0; i < H; i++) {
        for (int j = 0; j < W; j++)
            printf("%d ", img[i][j]);
        printf("\n");
    }
    
    #pragma omp parallel for
    for (int i = 0; i < H; i++) {
        for (int j = 0; j < W; j++) {
            img[i][j] += 10;
        }
    }

    printf("Pixel values after operation:\n");
    for (int i = 0; i < H; i++) {
        for (int j = 0; j < W; j++)
            printf("%d ", img[i][j]);
        printf("\n");
    }

    return 0;
}
