#include <stdio.h>
#include <omp.h>

#define N 10   // keep small for readable intermediate output

int main() {
    int a[N], b[N];
    int dot = 0;

    // Initialize vectors
    for (int i = 0; i < N; i++) {
        a[i] = i + 1;          // 1,2,3,...
        b[i] = (i + 1) * 2;    // 2,4,6,...
    }

    printf("Vector A and B values:\n");
    for (int i = 0; i < N; i++) {
        printf("i=%d, A=%d, B=%d\n", i, a[i], b[i]);
    }

    // Parallel dot product with reduction
    #pragma omp parallel for reduction(+:dot)
    for (int i = 0; i < N; i++) {
        int product = a[i] * b[i];
        dot += product;

        // Intermediate step print
        #pragma omp critical
        {
            printf("Thread %d -> i=%d, A[i]=%d, B[i]=%d, product=%d, thread-local partial dot=%d\n",
                   omp_get_thread_num(), i, a[i], b[i], product, dot);
        }
    }

    printf("Final Dot Product = %d\n", dot);
    return 0;
}
